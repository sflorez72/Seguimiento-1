#¿Cuántas features tiene el archivo?

#¿Cuáles son las primeras 10 features del organismo?
cat Stegastes_partitus.Stegastes_partitus-1.0.2.114.gff3 | cut -f3 | grep -v "#" | head


#¿Cuál es el top 10 de tipo de features (columna 3 más anotados en el genoma?
cat Stegastes_partitus.Stegastes_partitus-1.0.2.114.gff3 | cut -f3 | grep -v "#" | sort | uniq -c | sort -nr | head

#cut -f3 -> Muestra solo la columna 3
#grep -v -> Quita todos las entradas que empiece con #
#sort -> Ordena la lista de manera alfabética
#uniq -c -> Cuenta cuántas veces aparece cada tipo - deben de estar ya ordenada
